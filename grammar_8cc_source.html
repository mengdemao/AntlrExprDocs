25 {
26  // 生成解析字符串
27  ANTLRInputStream input_stream(input_string);
28 
29  // 词法解析
30  ExprLexer lexer(&input_stream);
31 
32  // 分割单词
33  CommonTokenStream tokens(&lexer);
34 
35  // 解析语法单元
36  ExprParser parser(&tokens);
37 
38  // 生成语法树
39  ParseTree* tree = parser.prog();
40 
41  // 检查词法分析和语法分析
42  if (lexer.getNumberOfSyntaxErrors() > 0 ||
43  parser.getNumberOfSyntaxErrors() > 0) {

52

55  visitor.visit(tree);
56 
57  return GRAMMAR_SUCCESS;
