253 {
254  proto_value sym1 = 0, sym2 = 0;
255  proto_value res = 0;
256 
257  switch (pcode.code) {
258  case OP_NOP:
259  break;
260 
261  case OP_PSH:
262  res = pcode.data;
263  psh(res);
264  break;
265 
266  case OP_POP:
267  (void)pop();
268  break;
269 
270  case OP_LDR:
271  get_var(pcode.data, res);
272  break;
273 
274  case OP_STR:
275  res = pop();
276  set_var(pcode.data, res);
277  break;
278 
279  case OP_ADD:
280  sym1 = pop();
281  sym2 = pop();
282  res = sym1 + sym2;
283  break;
284 
285  case OP_SUB:
286  sym1 = pop();
287  sym2 = pop();
288  res = sym1 - sym2;
289  psh(res);
290  break;
291 
292  case OP_MUL:
293  sym1 = pop();
294  sym2 = pop();
295  res = sym1 * sym2;
296  psh(res);
297  break;
298 
299  case OP_DIV:
300  sym1 = pop();
301  sym2 = pop();
302  res = sym1 / sym2;
303  psh(res);
304  break;
305 
306  case OP_MOD:
307  sym1 = pop();
308  sym2 = pop();
309  res = sym1 % sym2;
310  psh(res);
311  break;
312 
313  case OP_POW:

317  psh(res);
