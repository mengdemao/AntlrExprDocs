78 size_t MurmurHash::update(size_t hash, size_t value) {
79  static const size_t c1 = 0xCC9E2D51;
80  static const size_t c2 = 0x1B873593;
81 
82  size_t k1 = value;
83  k1 *= c1;
84  k1 = ROTL32(k1, 15);
85  k1 *= c2;
86 
87  hash ^= k1;
88  hash = ROTL32(hash, 13);
89  hash = hash * 5 + 0xE6546B64;
90 
91  return hash;
92 }
